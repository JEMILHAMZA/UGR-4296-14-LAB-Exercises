fun main() {
    val side1 = 6.0
    val side2 = 5.0
    val side3 = 7.0

    
    val triangleType = getTriangleType(side1, side2, side3)
    
    println("The triangle with sides $side1, $side2, and $side3 is $triangleType")
}

fun getTriangleType(side1: Double, side2: Double, side3: Double): String {
    return when {
        side1 == side2 && side2 == side3 -> "Equilateral"
        side1 == side2 || side2 == side3 || side1 == side3 -> "Isosceles"
        else -> "Scalene"
    }
}
