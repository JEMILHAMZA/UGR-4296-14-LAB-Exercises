

fun main() {
    val month=1
    

    
    val season= getSeason(month)
    
    println("The season is $season")
}

fun getSeason(month:Int): String {
    if (month==4 || month==5 ||month==6 ){
        return "Spring"
    }
    
    else if(month==7 || month==8 || month==9){
        
         return "Summer"
    }
    
     else if(month==10 || month==11 ||month==12){
        
         return "Autumn"
    }
     else{
        
         return "Winter"
    }
    
}
